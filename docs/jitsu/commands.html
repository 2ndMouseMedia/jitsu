commands.run = function (command, callback) {
  var parts = commands.parse(command),
      name = parts.shift(),
      action = parts.shift(),
      expected, resource;
  
  if (!commands.commands[name]) {
    return callback(new Error('Cannot run command on unknown resource: ' + name));
  }
